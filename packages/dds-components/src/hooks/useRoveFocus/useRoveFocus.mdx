<Source
  code={`
 import { useRoveFocus } from '@norges-domstoler/dds-components';
 import elements from './elements';
 import RoveItem from './RoveItem';

function MyComponent() {
const [focusedIndex, setFocus] = useRoveFocus(elements.length);

return (

<ul>
  {elements.map((element, index) => (
    <li key={element}>
      <RoveItem
        index={index}
        shouldFocus={focusedIndex === index}
        setFocus={setFocus}
      >
        {element.name}
      </RoveItem>
    </li>
  ))}
</ul>
) }; `} />
