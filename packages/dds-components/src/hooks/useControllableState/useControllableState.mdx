## Bruk

<Source code={`
import { useControllableState } from '@norges-domstoler/dds-components';

const MyComponent = ({
onChange,
checked: checkedProp,
defaultChecked
}) => {

const [checked, setChecked] = useControllableState({
value: checkedProp,
defaultValue: defaultChecked ?? false,
onChange,
});

return (

<input
  checked={checked}
  onChange={e => setChecked(e.target.checked)}
  type="checkbox"
/>
) };`} />
