  code={`
const [activeStep, setActiveStep] = useState(0);
const [completedSteps, setCompletedSteps] = useState(new Set());

function handleCompleted(index: number) {
// din kode for fullført steg
}

<ProgressTracker
  activeStep={activeStep}
  onStepChange={step => setActiveStep(step)}
>
  <ProgressTracker.Item onClick={handleCompleted}>Steg 1</ProgressTracker.Item>
  <ProgressTracker.Item onClick={handleCompleted}>Steg 2</ProgressTracker.Item>
  <ProgressTracker.Item onClick={handleCompleted}>
    Sammendrag
  </ProgressTracker.Item>
</ProgressTracker>

<div>
  {activeStep === 0 && (
    <div>
      Steg 1 <Button onClick={handleCompleted}> Neste </Button>
    </div>
  )}
  {activeStep === 1 && (
    <div>
      Steg 2 <Button onClick={handleCompleted}> Neste </Button>
    </div>
  )}
  {activeStep === 3 && <div>Sammendrag</div>}
</div>
`} />
